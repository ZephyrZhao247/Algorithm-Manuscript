## C++ V1

```C++
class Solution {
public:
    int rob(vector<int>& nums) {
      if (nums.size() == 0) return 0;
      if (nums.size() == 1) return nums[0];
      std::vector<int> robbed(nums.size(), 0);
      std::vector<int> not_robbed(nums.size(), 0);
      robbed[0] = nums[0];
      robbed[1] = nums[1];
      not_robbed[1] = nums[0];
      for (int i = 2; i < nums.size(); i++) {
        robbed[i] = nums[i] + not_robbed[i - 1];
        not_robbed[i] = std::max(robbed[i - 1], not_robbed[i - 1]);
      }
      return std::max(robbed[nums.size() - 1], not_robbed[nums.size() - 1]);
    }
};

```

Time Complexity: $O(n)$.

Space Complexity: $O(n)$.


## C++ V2

But we don't actually need 2 arrays in the V1.

```C++
class Solution {
public:
    int rob(vector<int>& nums) {
      if (nums.empty()) return 0;
      if (nums.size() == 1) return nums[0];
      std::vector<int> dp(nums.size(), 0);
      dp[0] = nums[0];
      dp[1] = std::max(nums[0], nums[1]);
      for (int i = 2; i < nums.size(); i ++) {
        dp[i] = std::max(dp[i-1], dp[i-2] + nums[i]);
      }
      return dp.back();
    }
};
```

Time Complexity: $O(n)$.

Space Complexity: $O(n)$.

## C++ V2

We could further collapse the array, since we only need to maintain 2 values at a time.

```C++
class Solution {
public:
    int rob(vector<int>& nums) {
      if (nums.empty()) return 0;
      if (nums.size() == 1) return nums[0];
      int p = std::max(nums[0], nums[1]), pp = nums[0];
      int curr = p;
      for (int i = 2; i < nums.size(); i ++) {
        curr = std::max(p, pp + nums[i]);
        pp = p;
        p = curr;
      }
      return curr;
    }
};
```

Time Complexity: $O(n)$.

Space Complexity: $O(n)$.
