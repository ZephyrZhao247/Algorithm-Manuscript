```C++
class Solution {
public:
  int rob(vector<int> &nums) {
    if (nums.empty())
      return 0;
    if (nums.size() == 1)
      return nums[0];
    if (nums.size() == 2)
      return std::max(nums[0], nums[1]);
    return std::max(robRange(nums, 0, nums.size() - 2),
                    robRange(nums, 1, nums.size() - 1));
  }
  int robRange(vector<int> &nums, int start, int end) {
    int prev_prev = nums[start];
    int prev = std::max(nums[start], nums[start + 1]);
    int curr = prev;
    for (int i = start + 2; i <= end; i++) {
      curr = std::max(prev_prev + nums[i], prev);
      prev_prev = prev;
      prev = curr;
    }
    return curr;
  }
};
```

Time Complexity: $O(n)$.

Space Complexity: $O(1)$.
