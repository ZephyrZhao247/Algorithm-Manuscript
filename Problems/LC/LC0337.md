## C++ V1

```C++
class Solution {
public:
    int rob(TreeNode* root) {
        return dfs(root, true);
    }

    int dfs(TreeNode* root, bool root_could_be_robbed) {
      if (root == nullptr) return 0;
      if (root_could_be_robbed) {
        return std::max(
          // Rob root
          root->val + dfs(root->left, false) + dfs(root->right, false),
          // Not rob root
          dfs(root->left, true) + dfs(root->right, true)
        );
      } else {
        return dfs(root->left, true) + dfs(root->right, true);
      }
    }
};
```

Too slow.

Time Complexity: $O(n^2)$.

Space Complexity: $O(log n)$.

## C++ V2

Use only 1 dfs for bottom-up state transfer.

Tree-shape DP.

```C++
class Solution {
public:
    int rob(TreeNode* root) {
      auto [rootRob, rootNotRob] = dfs(root);
      return std::max(rootRob, rootNotRob);
    }

    // return {couldRob, couldNotRob}
    std::tuple<int, int> dfs(TreeNode* root) {
      if (root == nullptr) return {0, 0};
      auto [leftRob, leftNotRob] = dfs(root->left);
      auto [rightRob, rightNotRob] = dfs(root->right);
      // dp[cur] = rob
      return {root->val + leftNotRob + rightNotRob, std::max(leftRob, leftNotRob) + std::max(rightRob, rightNotRob)};
    }
};
```

Time Complexity: $O(n)$.

Space Complexity: $O(log n)$.
